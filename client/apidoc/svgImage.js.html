goog.provide('xrx.svg.Image');



goog.require('xrx.svg.Element');
goog.require('xrx.svg.Namespace');



xrx.svg.Image = function() {};



xrx.svg.Image.tagName = 'image';



xrx.svg.Image.setHeight = xrx.svg.Element.setHeight;



xrx.svg.Image.setWidth = xrx.svg.Element.setWidth;



xrx.svg.Image.create = function(canvas, image) {
  var svgImage = xrx.svg.Element.create(xrx.svg.Image);
  svgImage.setAttributeNS(xrx.svg.Namespace['xlink'], 'xlink:href', image.src);
  return svgImage;
};
