goog.provide('xrx.canvas.Group');



xrx.canvas.Group = function() {

  this.childs_ = [];
};



xrx.canvas.Group.prototype.getChilds = function() {
  return this.childs_;
};



xrx.canvas.Group.addChild = function(group, child) {
  group.getChilds().push(child);
  child.draw();
};



xrx.canvas.Group.create = function(opt_properties) {
  return new xrx.canvas.Group();
};
