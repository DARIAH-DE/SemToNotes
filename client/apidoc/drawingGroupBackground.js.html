xrx.drawing.GroupBackground.prototype.create_ = function() {
  var graphics = this.getCanvas().getGraphics();
  var image = this.getCanvas().getImage();
  var backgroundImage = this.getCanvas().image_.element;
  this.element_ = graphics.Group.create();

  // install a black rectangle for panning
  this.panningRect_ = graphics.Rect.create(this.getCanvas());
  graphics.Rect.setWidth(this.panningRect_, this.getCanvas().getWidth());
  graphics.Rect.setHeight(this.panningRect_, this.getCanvas().getHeight());
  graphics.Rect.setFill(this.panningRect_, 'black');
  graphics.Group.addChild(this.element_, this.panningRect_);

  // install the background image
  backgroundImage = graphics.Image.create(this.getCanvas(), image);
  graphics.Image.setWidth(backgroundImage, image.naturalWidth);
  graphics.Image.setHeight(backgroundImage, image.naturalHeight);
  graphics.Group.addChild(this.element_, backgroundImage);

  // install a shield
  this.shield_ = new xrx.drawing.Shield(this.getCanvas());
  goog.dom.append(this.element_, this.shield_.getElement());
