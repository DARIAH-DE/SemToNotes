xrx.canvas.GroupBackground.handleMouseDown = function(e, canvas) {
  if (canvas.mode_ !== xrx.canvas.Mode.PAN) return;
  var eventPoint = [e.clientX, e.clientY];
  var transform = new goog.math.AffineTransform();

  if (!canvas.stateOrigin) canvas.stateOrigin = new Array(2);

  transform.createInverse().transform(eventPoint, 0, canvas.stateOrigin, 0, 1);

  canvas.state = xrx.canvas.State.DRAG;

xrx.canvas.GroupBackground.handleMouseMove = function(e, canvas) {
  if (canvas.mode_ !== xrx.canvas.Mode.PAN) return;
  if (canvas.state !== xrx.canvas.State.DRAG) return;

  var point = new Array(2);
  var eventPoint = [e.clientX, e.clientY];
  var transform = new goog.math.AffineTransform();

  transform.createInverse().transform(eventPoint, 0, point, 0, 1);

  var x = point[0] - canvas.stateOrigin[0];
  var y = point[1] - canvas.stateOrigin[1];

  canvas.getViewBox().ctm = transform.translate(x, y).concatenate(canvas.getViewBox().ctm);

  canvas.setCTM(canvas.getViewBox().ctm);

  canvas.stateOrigin = point;

xrx.canvas.GroupBackground.prototype.create_ = function() {
  var image = this.getCanvas().getImage();
  this.element_ = xrx.svg.Group.create();

  // install a black rectangle for panning
  var rect = xrx.svg.Element.createNS('rect');
  xrx.svg.Element.setProperties(rect, {
    'width': '100%',
    'height': '100%'
  });
  goog.dom.append(this.element_, rect);

  // install the background image
  image = this.getCanvas().image_.element = xrx.svg.Image.create({
    'xlink:href': image.getAttribute('src'),
    'width': image.naturalWidth,
    'height': image.naturalHeight
  });
  goog.dom.append(this.element_, image);

  // install a shield
  this.shield_ = new xrx.canvas.Shield(this.getCanvas());
  goog.dom.append(this.element_, this.shield_.getElement());
