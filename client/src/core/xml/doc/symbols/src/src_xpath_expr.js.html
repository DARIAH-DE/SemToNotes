152   var exrs = this.evaluate(ctx);

176 ***REMOVED*** Evaluate and interpret the result as a boolean value.
177 ***REMOVED***
178 ***REMOVED*** @param {!xrx.xpath.Context} ctx The context to evaluate the expression in.
179 ***REMOVED*** @return {boolean} The evaluated boolean value.
180***REMOVED*****REMOVED***

182   var exrs = this.evaluate(ctx);
183   if (exrs instanceof xrx.xpath.NodeSet) {
