105 ***REMOVED***
106 ***REMOVED*** Returns whether the cursor caret or the left edge of a cursor
107 ***REMOVED*** selection is placed at the very end of a WYSIWYM control.
108 ***REMOVED***
109 ***REMOVED*** @param {!xrx.richxml} wysiwym The WYSIWYM control.
110 ***REMOVED*** @return {boolean} True if the cursor is placed at the end
111 ***REMOVED*** otherwise false.
112***REMOVED*****REMOVED***
113 xrx.wysiwym.cursor.leftAtEndPosition = function(wysiwym) {
114   var pos = xrx.wysiwym.cursor.leftPosition(wysiwym);
115   var cm = wysiwym.codemirror_;
116   var last = cm.lineCount() - 1;
117   var line = cm.getLine(last);
118 
119   return pos.line === last && pos.ch === line.length;
120 };
